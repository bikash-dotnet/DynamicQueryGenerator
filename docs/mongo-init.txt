
### `scripts/mongo-init.js` (For Docker initialization)
```javascript
// MongoDB initialization script for Docker

// Switch to target database
db = db.getSiblingDB(process.env.MONGO_INITDB_DATABASE || 'nlp_query_system');

// Create collections
db.createCollection('user_queries');
db.createCollection('query_logs');

// Create indexes for user_queries
db.user_queries.createIndex({ query_hash: 1 }, { unique: true });
db.user_queries.createIndex({ created_at: 1 }, { expireAfterSeconds: 2592000 });
db.user_queries.createIndex({ collection_name: 1, last_used: -1 });
db.user_queries.createIndex({ original_text: "text" });

// Create indexes for query_logs
db.query_logs.createIndex({ timestamp: -1 });
db.query_logs.createIndex({ user_ip: 1, timestamp: -1 });
db.query_logs.createIndex({ was_rejected: 1 });

// Insert sample data (for testing)
db.users.insertMany([
    { name: "John Doe", email: "john@example.com", age: 35, country: "USA", status: "active", created_at: new Date() },
    { name: "Jane Smith", email: "jane@example.com", age: 28, country: "UK", status: "active", created_at: new Date() },
    { name: "Raj Patel", email: "raj@example.com", age: 42, country: "India", status: "active", created_at: new Date() }
]);

db.products.insertMany([
    { name: "Laptop", category: "electronics", price: 999.99, stock: 50, brand: "TechBrand", created_at: new Date() },
    { name: "Mouse", category: "electronics", price: 29.99, stock: 200, brand: "TechBrand", created_at: new Date() },
    { name: "Book", category: "books", price: 19.99, stock: 100, brand: "Publisher", created_at: new Date() }
]);

print("MongoDB initialization completed!");
